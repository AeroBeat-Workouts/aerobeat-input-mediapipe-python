## Executive Summary

As of February 2026, **MediaPipe offers robust native SDK support for both Android and iOS platforms** through the MediaPipe Tasks API. Unlike the Python-based desktop implementation currently used in AeroBeat, mobile platforms utilize optimized native libraries (AAR for Android, XCFramework for iOS) that provide hardware-accelerated inference with significantly better performance on mobile devices.

**Key Findings:**
- MediaPipe Tasks API provides official Android (Java/Kotlin) and iOS (Swift/Objective-C) SDKs
- Mobile performance: 30+ FPS on mid-range devices, 60 FPS on flagship devices
- Godot 4.6 requires custom Android/iOS plugins for camera + MediaPipe integration
- JNI bridge or Objective-C++ wrapper needed for GDExtension integration
- Camera permissions and lifecycle management are platform-specific concerns

### 2.2 Hand Landmarker Implementation

**Key Classes:**
```kotlin
import com.google.mediapipe.tasks.vision.handlandmarker.HandLandmarker
import com.google.mediapipe.tasks.vision.handlandmarker.HandLandmarkerResult
import com.google.mediapipe.tasks.vision.core.RunningMode
import com.google.mediapipe.tasks.core.BaseOptions

// Initialize
val baseOptions = BaseOptions.builder()
    .setModelAssetPath("hand_landmarker.task")
    .setDelegate(Delegate.GPU) // or Delegate.CPU
    .build()

val options = HandLandmarker.HandLandmarkerOptions.builder()
    .setBaseOptions(baseOptions)
    .setRunningMode(RunningMode.LIVE_STREAM)
    .setNumHands(2) // Two-hand detection for boxing
    .setMinHandDetectionConfidence(0.5f)
    .setMinHandPresenceConfidence(0.5f)
    .setMinTrackingConfidence(0.5f)
    .setResultListener(this::onHandLandmarkerResult)
    .setErrorListener(this::onHandLandmarkerError)
    .build()

val handLandmarker = HandLandmarker.createFromOptions(context, options)

// Process frame
val mpImage = BitmapImageBuilder(bitmap).build()
handLandmarker.detectAsync(mpImage, timestampMs)
```

### 2.3 Camera Integration on Android

**CameraX (Recommended for Godot Integration):**
```kotlin
import androidx.camera.core.CameraSelector
import androidx.camera.core.ImageAnalysis
import androidx.camera.core.ImageProxy
import androidx.camera.lifecycle.ProcessCameraProvider
import java.util.concurrent.Executors

val cameraExecutor = Executors.newSingleThreadExecutor()

val imageAnalysis = ImageAnalysis.Builder()
    .setBackpressureStrategy(ImageAnalysis.STRATEGY_KEEP_ONLY_LATEST)
    .setOutputImageFormat(ImageAnalysis.OUTPUT_IMAGE_FORMAT_RGBA_8888)
    .build()

imageAnalysis.setAnalyzer(cameraExecutor) { imageProxy ->
    val bitmap = imageProxy.toBitmap() // Convert to Bitmap
    val mpImage = BitmapImageBuilder(bitmap).build()
    handLandmarker.detectAsync(mpImage, imageProxy.imageInfo.timestamp)
    imageProxy.close()
}

val cameraSelector = CameraSelector.DEFAULT_FRONT_CAMERA

// Bind to lifecycle
val cameraProvider = ProcessCameraProvider.getInstance(context).await()
cameraProvider.bindToLifecycle(lifecycleOwner, cameraSelector, imageAnalysis)
```

### 3.2 Hand Landmarker Implementation (Swift)

```swift
import MediaPipeTasksVision

class HandTracker: NSObject {
    private var handLandmarker: HandLandmarker?
    
    func setupHandLandmarker() throws {
        let baseOptions = BaseOptions()
        baseOptions.modelAssetPath = "hand_landmarker.task"
        
        // Use GPU delegate for better performance
        baseOptions.delegate = .GPU
        
        let options = HandLandmarkerOptions()
        options.baseOptions = baseOptions
        options.runningMode = .liveStream
        options.numHands = 2
        options.minHandDetectionConfidence = 0.5
        options.minHandPresenceConfidence = 0.5
        options.minTrackingConfidence = 0.5
        
        handLandmarker = try HandLandmarker(options: options)
    }
    
    func processFrame(_ sampleBuffer: CMSampleBuffer) {
        let image = MPImage(sampleBuffer: sampleBuffer)
        try? handLandmarker?.detectAsync(image: image, timestampInMilliseconds: timestamp)
    }
}
```

#include <gdextension_interface.h>
#include <godot_cpp/core/class_db.hpp>

class MediaPipeProvider : public Node {
    GDCLASS(MediaPipeProvider, Node)
    
protected:
    static void _bind_methods();

    void startTracking();
    void stopTracking();
    Array getHandLandmarks();

## 7. Recommended Architecture

### Cross-Platform Input Abstraction

```gdscript
# Godot GDScript interface
class_name MobileMediaPipeInput
extends Node

@export var enable_hand_tracking: bool = true
@export var enable_pose_tracking: bool = true
@export var camera_facing: CameraFacing = CameraFacing.FRONT

enum CameraFacing { FRONT, BACK }

# Platform-specific implementation
var _provider: MediaPipeProvider

func _ready():
    if OS.has_feature("android"):
        _provider = preload("res://addons/mediapipe/android_provider.gd").new()
    elif OS.has_feature("ios"):
        _provider = preload("res://addons/mediapipe/ios_provider.gd").new()
    else:
        push_warning("MobileMediaPipeInput only works on Android/iOS")
        return
    
    add_child(_provider)

func start_tracking():
    _provider.startTracking()

func get_hand_landmarks(hand_index: int) -> Array:
    return _provider.getHandLandmarks(hand_index)

func get_pose_landmarks() -> Array:
    return _provider.getPoseLandmarks()
```

- Native performance on mobile devices
- Hardware acceleration (GPU/Neural Engine)
